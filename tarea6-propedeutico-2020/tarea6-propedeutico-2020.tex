\section*{Resolver las siguientes ecuaciones:}

\begin{enumerate}
    \item $(x + 4)3 - (x - 3)3 = 343$
    \item $\dfrac{x+3}{2x-1} - \dfrac{5x-1}{4x+7} = 0$
    \item $3(2x^2 - mx)+4nx - 2mn=0$
    \item $\dfrac{5}{2x2} - \dfrac{1}{6x^2} = \dfrac{7}{12}$
    \item $\sqrt{2x-1}+\sqrt{x+3}=3$
\end{enumerate}
